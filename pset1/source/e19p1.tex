\begin{align*}
    \boldsymbol{v} = \begin{bmatrix}
    x \\ y \\ z
    \end{bmatrix}
\end{align*}

Find a $3 \times 3$ matrix $\boldsymbol{M}$ such that $\boldsymbol{Mv} = 3\boldsymbol{v}$ for any vector v.

\begin{solution}
\begin{align*}
    \boldsymbol{M} = \begin{bmatrix}
    3 & 0 & 0 \\ 0 & 3 & 0 \\ 0 & 0 & 3
    \end{bmatrix}
\end{align*}
\end{solution}
