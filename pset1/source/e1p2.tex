Compute $\boldsymbol{v} \cdot \boldsymbol{w}$ where

\begin{center}
    $\boldsymbol{v} = 
    \begin{bmatrix}
        1 \\ 3 \\ -4 \\ 6
    \end{bmatrix}$
    , and $\boldsymbol{w} = \begin{bmatrix}
        -2 \\ 0 \\ 1 \\ 3
    \end{bmatrix}$
\end{center}

\begin{solution}

\begin{align*}
\boldsymbol{v} \cdot \boldsymbol{w} = \|\boldsymbol{v}\| \|\boldsymbol{w}\| cos(\theta) = 12
\end{align*}

\end{solution}
