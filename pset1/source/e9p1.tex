Evaluate $4\boldsymbol{A} - 5\boldsymbol{B}$

\begin{align*}
    \boldsymbol{A} = \begin{bmatrix}
    3 & 4 & 3 \\
    3 & 1 & 1
    \end{bmatrix}\\
    \boldsymbol{B} = \begin{bmatrix}
    1 & 2 & 3 \\
    2 & 2 & 1
    \end{bmatrix}
\end{align*}

\begin{solution}
\begin{align*}
    4\boldsymbol{A} - 5\boldsymbol{B} = \begin{bmatrix}
    7 & 6 & -3 \\
    2 & -6 & -1
    \end{bmatrix}
\end{align*}
\end{solution}
