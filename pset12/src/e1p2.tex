Determine (by hand) the unit tangent vector.

\begin{solution}
\begin{align*}
    \boldsymbol{r'}(u) &= -R\sin u \hat{\boldsymbol{i}} + R\cos u \hat{\boldsymbol{j}} \\
    \boldsymbol{\hat{T}} &= \frac{r'}{|r'|} \\
    &= -\sin u \hat{\boldsymbol{i}} + \cos u \hat{\boldsymbol{j}}
\end{align*}
\end{solution}
