Write the appropriate equations for find the unknown forces and torques acting on the beam.

\begin{solution}
\begin{align*}
    \sum F_x = 0 &= F_{Nx} + F_r\cos{\theta} \\
    \sum F_y = 0 &= -F_{Ny} + F_g - F_r\sin{\theta} \\
    \sum \tau_z = 0 &= F_g\frac{L}{2} - F_r\sin{\theta}L
\end{align*}
\end{solution}
