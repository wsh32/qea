Rewrite the equations above in the form of a rotation matrix.

\begin{solution}
\begin{align*}
    \begin{bmatrix}\hat{x} \\ \hat{y}\end{bmatrix} =
    \begin{bmatrix}
        \cos\theta & -\sin\theta \\
        \sin\theta & \cos\theta
    \end{bmatrix} \begin{bmatrix}
        \hat{i} \\ \hat{j}
    \end{bmatrix}
\end{align*}
\end{solution}
