Load three face images, and calculate the distance between each pair.

\begin{solution} \
    \begin{lstlisting}[language=Matlab]
load('face_bases.mat');
im1 = test_images(:,:,1);
im2 = test_images(:,:,2);
im3 = test_images(:,:,3);

dist_im12 = sqrt(sum(sum(((im1-im2).^2))))
dist_im23 = sqrt(sum(sum(((im2-im3).^2))))
dist_im13 = sqrt(sum(sum(((im1-im3).^2))))
    \end{lstlisting}
