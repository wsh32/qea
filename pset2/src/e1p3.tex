Calculate the determinant of the matrix that shears in the horizontal direction.

\begin{align*}
    \begin{bmatrix}
        1 & 1 \\
        0 & 1
    \end{bmatrix}
\end{align*}

\begin{solution}
\begin{align*}
    \det \left( \begin{bmatrix}
        1 & 1 \\
        0 & 1
    \end{bmatrix} \right) = 1
\end{align*}
\end{solution}
