\begin{align*}
    \begin{bmatrix}
        -1 & 0 \\
        0 & 1
    \end{bmatrix}
\end{align*}

\begin{solution}
    \begin{align*}
        \left(\begin{bmatrix}
            -1 & 0 \\
            0 & 1
        \end{bmatrix}\right)^{-1}
        = \begin{bmatrix}
            1 & 0 \\
            0 & -1
        \end{bmatrix}
    \end{align*}
\end{solution}
