\begin{align*}
    \int_{0}^{1} \int_{y}^{\exp{y}} dx dy
\end{align*}

\begin{solution} \
    \begin{center}
        \includegraphics[width=0.375\textwidth]{img/e2p2.png}
    \end{center}
    
    \begin{align*}
        \int_{0}^{1} \int_{y}^{\exp{y}} dx dy &= \int_{0}^{1} (\exp{y} - y) dy\\
        &= \exp{y}\bigg|_{0}^{1} - \frac{y^2}{2}\bigg|_{0}^{1} \\
        &= e - \frac{3}{2}
    \end{align*}
\end{solution}
