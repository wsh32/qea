\begin{align*}
    \int_{1}^{2} \int_{0}^{\ln{x}} dy dx
\end{align*}

\begin{solution} \
    \begin{center}
        \includegraphics[width=0.375\textwidth]{img/e1p2.png}
    \end{center}
    
    \begin{align*}
        \int_{1}^{2} \int_{0}^{\ln{x}} dy dx &= \int_{1}^{2} \ln{x} dx \\
        &= \left(x\ln{x} - x\right) \bigg|_{0}^{1} \\
        &= 2 \ln{2} - 1
    \end{align*}
\end{solution}
