\begin{align*}
    \int_{0}^{1} \int_{y}^{2-y} dx dy
\end{align*}

\begin{solution} \
    \begin{center}
        \includegraphics[width=0.375\textwidth]{img/e2p1.png}
    \end{center}
    
    \begin{align*}
        \int_{0}^{1} \int_{y}^{2-y} dx dy &= \int_{0}^{1} (2-2y) dy \\
        &= \left(2y - y^2\right)\bigg|_{0}^{1} \\
        &= 1
    \end{align*}
\end{solution}
