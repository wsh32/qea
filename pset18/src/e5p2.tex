\begin{align*}
    \boldsymbol{F}(x, y, z) = y\sin{z} \boldsymbol{\hat{i}} - x\sin{z} \boldsymbol{\hat{j}} + \cos{z} \boldsymbol{\hat{k}}
\end{align*}

\begin{solution}
\begin{align*}
    \mathrm{div}\boldsymbol{F} &= \frac{\partial X}{\partial x} + \frac{\partial Y}{\partial y} + \frac{\partial Z}{\partial z} \\
    &= -\sin{z}
\end{align*}
\end{solution}
