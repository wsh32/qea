Find $\frac{dz}{dt}$.

\begin{align*}
    z &= x^2 + y^2 + xy, x = \sin(t), y=e^t \\
\end{align*}

\begin{solution}
\begin{align*}
    \frac{dz}{dt} &= (2x+y)(-\cos(t)) + (2y+x)\left(e^t\right) \\
    \frac{dz}{dt} &= -2x\cos(t) - y\cos(t) + 2ye^t + xe^t \\
    \frac{dz}{dt} &= x\left(e^t - 2\cos(t)\right) + y\left(2e^t - \cos(t)\right)
\end{align*}
\end{solution}
