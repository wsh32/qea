\begin{align*}
    x_1 + x_2 + x_3 &= 6 \\
    x_2 + x_3 &= 2 \\
    x_1 - 2x_3 &= 4
\end{align*}

\begin{solution}
    \begin{align*}
        \begin{bmatrix}
            6 \\ 2 \\ 4
        \end{bmatrix} &= \begin{bmatrix}
            1 & 1 & 1 \\
            0 & 1 & 1 \\
            1 & 0 & -2
        \end{bmatrix}\begin{bmatrix}
            x_1 \\ x_2 \\ x_3
        \end{bmatrix} \\
        \begin{bmatrix}
            x_1 \\ x_2 \\ x_3
        \end{bmatrix} &= \begin{bmatrix}
            4 \\ 2 \\ 0
        \end{bmatrix}
    \end{align*}
\end{solution}
