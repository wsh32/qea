    x_1 + 2x_3 &= 4
\end{align*}

            1 & 1 & 1 \\

            x_1 \\ x_2 \\ x_3
        \end{bmatrix} \\

            x_1 \\ x_2 \\ x_3

        \end{bmatrix}
