        0 & 0 & 1 \\

        0.5 & 0.866 & 0.634 \\
        -0.866 & 0.5 & -3.098 \\
        0 & 0 & 1
    \end{bmatrix} \\
    r_M &= R_{MG} T_{MG} r_G \\
    &= \begin{bmatrix}
        0.7679 \\
       -5.3301 \\
        1
    \end{bmatrix} \\
    r_M &= 0.7679 \boldsymbol{\hat{i}}_M - 5.3301 \boldsymbol{\hat{j}}_M
\end{align*}
